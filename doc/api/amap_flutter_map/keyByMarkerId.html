Map<String, Marker> keyByMarkerId(Iterable<Marker> markers) {
  if (markers == null) {
    return <String, Marker>{};
  }
  return Map<String, Marker>.fromEntries(markers.map(
      (Marker marker) => MapEntry<String, Marker>(marker.id, marker.clone())));
}
